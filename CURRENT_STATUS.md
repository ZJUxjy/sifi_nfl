## 🎮 现在可以做什么？

### ✅ 可以运行的代码

```typescript
// 1. 生成完整的联赛
import { generateAllTeams } from './src/worker/core/team';
const { teams, players } = generateAllTeams(2025);
console.log(`生成了 ${teams.length} 支球队, ${players.length} 名球员`);

// 2. 模拟单场比赛
import { GameSim } from './src/worker/core/game';
const team1 = teams[0];
const team2 = teams[1];
const game = new GameSim({
  gid: 1,
  teams: [team1, team2],
  quarterLength: 15,
});
const result = game.run();
console.log(`比分: ${result.teams[0].pts} - ${result.teams[1].pts}`);

// 3. 模拟整个赛季
import { SeasonManager } from './src/worker/core/season';
const season = new SeasonManager(2025, teams);
season.startRegularSeason();

for (let week = 1; week <= 17; week++) {
  const games = season.simWeek();
  console.log(`第${week}周: ${games.length}场比赛`);
}

// 4. 查看积分榜
const standings = season.getStandings();
console.log('积分榜:', standings.slice(0, 10));
```
